\def\TCID{TC-0001}

\def\TCTitle{Usage/version check}

\def\TCPurpose{
	Verify that wpm executes, exits cleanly, and reports its bundled and Windows system-library dependency versions.
}

\def\TCPriority{\TCPriorityHigh}

\def\TCRequirementRef{REQ-0001}

\def\TCPreconditions{
	None.
}

\def\TCEnvironment{
	Local Windows developer workstation. Future target: Windows Container.
}

\def\TCInputData{
	No input required.
}

\def\TCProcedure{
	\par
	\begin{enumerate}
		\item Run \texttt{wpm.exe}
		\item Run \texttt{wpm.exe --version}
	\end{enumerate}
}

\def\TCExpectedResult{
	\par
	\begin{enumerate}
		\item Exit code 0 and version or usage information is displayed.
		\item Exit code 0 and dependency version information is displayed for miniz, libsodium, urlmon, and advapi32.
	\end{enumerate}
}

\def\TCPostConditions{
	Test environment remains operational.
}
\TCTemplate
