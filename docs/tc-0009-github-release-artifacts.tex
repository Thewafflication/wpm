\def\TCID{TC-0009}

\def\TCTitle{GitHub Release artifact workflow}

\def\TCPurpose{
	Verify that the tag-triggered GitHub workflow uses the protected durable release key and validates each signed Release package before publication.
}

\def\TCPriority{\TCPriorityHigh}

\def\TCRequirementRef{REQ-0009}

\def\TCPreconditions{
	The source repository contains the GitHub Release workflow.
}

\def\TCEnvironment{
	Local Windows developer workstation or GitHub Actions Windows runner.
}

\def\TCInputData{
	The checked-in \texttt{.github/workflows/release.yml} workflow definition.
}

\def\TCProcedure{
	\par
	\begin{enumerate}
		\item Validate that the Release workflow is triggered by tags.
		\item Validate that it builds all Release architectures.
		\item Validate that x86 Debug WPM packages the Release executables into ZIP assets.
		\item Validate that packaging downloads the successful x86 verification artifact and does not rebuild x86 Debug.
		\item Validate that the protected release environment supplies the durable private key without publishing it.
		\item Validate that each finished package is verified with the checked-in durable public key before upload.
	\end{enumerate}
}

\def\TCExpectedResult{
	\par
	\begin{enumerate}
		\item The workflow configuration is valid.
		\item On a non-tag run, the test passes and records that release publication is not required.
		\item On a tag run, the Release workflow verifies, packages, and publishes the assets.
		\item Publication depends on successful signature and package-content validation of all three archives.
		\item The executable used as package builder is the same artifact that produced the x86 verification evidence.
	\end{enumerate}
}

\def\TCPostConditions{
	No release assets are created by non-tag test execution.
}
\TCTemplate
