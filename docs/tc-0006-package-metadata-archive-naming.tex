\def\TCID{TC-0006}

\def\TCTitle{Package metadata archive naming}

\def\TCPurpose{
	Verify that \texttt{wpm build} names release and debug archives from
	\texttt{.wpm/package.txt} metadata.
}

\def\TCPriority{\TCPriorityHigh}

\def\TCRequirementRef{REQ-0006}

\def\TCPreconditions{
	A writable temporary source directory and output directory are available.
}

\def\TCEnvironment{
	Local Windows developer workstation. Future target: Windows Container.
}

\def\TCInputData{
	A source package with \texttt{name}, \texttt{version}, \texttt{arch}, and
	\texttt{debug} metadata.
}

\def\TCProcedure{
	\par
	\begin{enumerate}
		\item Create a temporary source package directory.
		\item Write package metadata with \texttt{debug=false}.
		\item Run \texttt{wpm.exe build <source-dir> <output-dir> --no-index}.
		\item Write package metadata with \texttt{debug=true}.
		\item Run \texttt{wpm.exe build <source-dir> <output-dir> --no-index}.
	\end{enumerate}
}

\def\TCExpectedResult{
	\par
	\begin{enumerate}
		\item Release build exits 0 and writes
		\texttt{<name>-<version>-<arch>.zip}.
		\item Debug build exits 0 and writes
		\texttt{<name>-<version>-<arch>-debug.zip}.
	\end{enumerate}
}

\def\TCPostConditions{
	Temporary source and output directories are removed.
}
\TCTemplate
