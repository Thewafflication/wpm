\def\TCID{TC-0004}

\def\TCTitle{Package archive installation lifecycle}

\def\TCPurpose{
	Verify that \texttt{wpm install} stages, verifies, installs, and stores a ZIP package.
}

\def\TCPriority{\TCPriorityHigh}

\def\TCRequirementRef{REQ-0004}

\def\TCPreconditions{
	A valid package archive exists and the WPM data directory is writable.
}

\def\TCEnvironment{
	Local Windows developer workstation. Future target: Windows Container.
}

\def\TCInputData{
	A package archive containing package metadata, \texttt{hello.txt}, and
	\texttt{nested/data.txt}.
}

\def\TCProcedure{
	\par
	\begin{enumerate}
		\item Create a temporary source package directory.
		\item Add \texttt{.wpm/package.txt} metadata.
		\item Build a package archive with \texttt{wpm.exe build}.
		\item Run \texttt{wpm.exe install <archive>}.
		\item Inspect the staged package behavior, package deployment result, and local archive store.
	\end{enumerate}
}

\def\TCExpectedResult{
	\par
	\begin{enumerate}
		\item Install command exits 0.
		\item Package is staged beneath \texttt{\%ProgramData\%/WPM/temp}.
		\item The package index is verified before \texttt{.wpm/install.cmd} executes.
		\item The install script's expected deployment result is present.
		\item The ZIP archive is retained beneath \texttt{\%ProgramData\%/WPM/packages}.
		\item The staging directory is removed.
	\end{enumerate}
}

\def\TCPostConditions{
	Temporary source, package output, and staging directories are removed; the
	stored package archive remains for package-management operations.
}
\TCTemplate
